# Payload CMS Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the file-based Markdown content system with Payload CMS embedded in the website app, giving non-technical editors an admin UI for content management.

**Architecture:** Payload 3.x embedded in the existing Next.js website app at `apps/website/`. Payload admin panel at `/admin`, REST API at `/api`. Frontend pages use Payload Local API (`getPayload()`) to fetch content. Two route groups: `(payload)/` for CMS and `(frontend)/` for public pages. Localization via Payload field-level `localized: true` for content, next-intl unchanged for UI strings.

**Tech Stack:** Payload CMS 3.x, `@payloadcms/db-postgres` (Neon), `@payloadcms/richtext-lexical`, `@payloadcms/next`, `@payloadcms/storage-vercel-blob`, Next.js 16, next-intl v4, pnpm monorepo.

**Spec:** `docs/superpowers/specs/2026-03-25-payload-cms-migration-design.md`

```ts
// apps/website/payload.config.ts
import { fileURLToPath } from "node:url"
import path from "path"
import { buildConfig } from "payload"

const __dirname = path.dirname(fileURLToPath(import.meta.url))
import { postgresAdapter } from "@payloadcms/db-postgres"
import { lexicalEditor } from "@payloadcms/richtext-lexical"
import { vercelBlobStorage } from "@payloadcms/storage-vercel-blob"
import { en } from "@payloadcms/translations/languages/en"
import { nl } from "@payloadcms/translations/languages/nl"
import { Categories } from "./collections/Categories"
import { Services } from "./collections/Services"
import { Guides } from "./collections/Guides"
import { LandingPages } from "./collections/LandingPages"
import { Pages } from "./collections/Pages"
import { Media } from "./collections/Media"

export default buildConfig({
  admin: {
    importMap: {
      baseDir: path.resolve(__dirname),
    },
  },
  db: postgresAdapter({
    pool: { connectionString: process.env.DATABASE_URL! },
  }),
  editor: lexicalEditor(),
  localization: {
    locales: [
      { label: { en: "English", nl: "Engels" }, code: "en" },
      { label: { en: "Dutch", nl: "Nederlands" }, code: "nl" },
    ],
    defaultLocale: "en",
    fallback: true,
  },
  i18n: {
    supportedLanguages: { en, nl },
    fallbackLanguage: "en",
  },
  collections: [Categories, Services, Guides, LandingPages, Pages, Media],
  plugins: [
    ...(process.env.BLOB_READ_WRITE_TOKEN
      ? [
          vercelBlobStorage({
            collections: { media: true },
            token: process.env.BLOB_READ_WRITE_TOKEN,
          }),
        ]
      : []),
  ],
  secret: process.env.PAYLOAD_SECRET!,
  typescript: {
    outputFile: path.resolve(__dirname, "payload-types.ts"),
  },
})
```

- `[locale]/layout.tsx` (locale layout)

- `[locale]/guides/` (guide pages)

```bash

mkdir -p "(frontend)"
mv [locale] "(frontend)/"
```

Also move `app/layout.tsx` or `app/[locale]/layout.tsx` appropriately — the root `app/layout.tsx` stays as-is (it's the global layout). Only the `[locale]` segment and its children move.

**Note:** Route groups are invisible in URLs, so all existing URLs continue to work unchanged.

**API route conflict check:** The Payload REST API at `(payload)/api/[...slug]` and the existing `api/search/` routes both resolve to `/api/...`. Next.js gives specific routes precedence over catch-all routes, so `/api/search` will correctly hit the existing route, not Payload's catch-all. Verify this works during testing.

Verify the app still builds and pages render correctly at the same URLs.

```ts
  const payload = await getPayload({ config })
  const reset = process.argv.includes("--reset")

  if (reset) {
    console.log("Resetting database...")
    // Delete in reverse dependency order
    // Use { id: { exists: true } } to match all documents
    const deleteAll = { id: { exists: true } } as any
    await payload.delete({ collection: "guides", where: deleteAll })
    await payload.delete({ collection: "landing-pages", where: deleteAll })
    await payload.delete({ collection: "pages", where: deleteAll })
    await payload.delete({ collection: "services", where: deleteAll })
    await payload.delete({ collection: "categories", where: deleteAll })
    await payload.delete({ collection: "media", where: deleteAll })
    console.log("Database cleared.")

  const categoryMap = await importCategories(payload)
  const serviceMap = await importServices(payload, categoryMap)
  await importGuides(payload, categoryMap, serviceMap)
  await importLandingPages(payload, categoryMap, serviceMap)
  await importPages(payload)

  console.log("Seed complete!")
  process.exit(0)

  console.error("Seed failed:", err)
```

```ts
// apps/website/seed/importServices.ts
import type { Payload } from "payload"
import {
  getServiceSlugs,
  getServiceBySlug,
} from "@switch-to-eu/content"
import type { Locale } from "@switch-to-eu/content"
import { markdownToLexical } from "./markdownToLexical"

// NOTE: getEUServices()/getNonEUServices() return flat ServiceFrontmatter[],
// NOT objects with .slug/.content. We use getServiceSlugs() + getServiceBySlug()
// to get the full data including content body.

export async function importServices(
  payload: Payload,
  categoryMap: Map<string, number>
): Promise<Map<string, number>> {
  const serviceMap = new Map<string, number>()

  // Get all service slugs for both regions
  const euSlugs = getServiceSlugs("eu", "en")
  const nonEuSlugs = getServiceSlugs("non-eu", "en")
  const allSlugs = [...euSlugs, ...nonEuSlugs]

  for (const slug of allSlugs) {
    // getServiceBySlug returns { frontmatter, content, segments }
    const enService = getServiceBySlug(slug, "en" as Locale)
    if (!enService) continue

    const enData = enService.frontmatter
    console.log(`Importing service: ${slug} (${enData.region})`)

    const enLexical = enService.content
      ? await markdownToLexical(enService.content)
      : undefined

    // Get Dutch version
    let nlService = null
    try {
      nlService = getServiceBySlug(slug, "nl" as Locale)
    } catch {
      // Dutch version may not exist
    }

    const nlLexical = nlService?.content
      ? await markdownToLexical(nlService.content)
      : undefined

    const categoryId = categoryMap.get(enData.category)

    const created = await payload.create({
      collection: "services",
      locale: "en",
      data: {
        name: enData.name,
        slug,
        category: categoryId,
        region: enData.region,
        location: enData.location,
        freeOption: enData.freeOption ?? false,
        startingPrice: enData.startingPrice ?? "",
        description: enData.description,
        url: enData.url,
        featured: enData.featured ?? false,
        features: enData.features?.map((f: string) => ({ feature: f })) ?? [],
        tags: enData.tags?.map((t: string) => ({ tag: t })) ?? [],
        content: enLexical,
        issues:
          enData.region === "non-eu" && enData.issues
            ? enData.issues.map((i: string) => ({ issue: i }))
            : [],
        // recommendedAlternative resolved in a second pass
      },
    })

    // Update Dutch locale
    if (nlService) {
      const nlData = nlService.frontmatter
      await payload.update({
        collection: "services",
        id: created.id,
        locale: "nl",
        data: {
          name: nlData.name,
          description: nlData.description,
          startingPrice: nlData.startingPrice ?? "",
          features:
            nlData.features?.map((f: string) => ({ feature: f })) ?? [],
          content: nlLexical,
          issues:
            nlData.region === "non-eu" && nlData.issues
              ? nlData.issues.map((i: string) => ({ issue: i }))
              : [],
        },
      })
    }

    serviceMap.set(slug, created.id)
  }

  // Second pass: resolve recommendedAlternative relationships
  for (const slug of allSlugs) {
    const enService = getServiceBySlug(slug, "en" as Locale)
    if (!enService) continue
    const fm = enService.frontmatter
    if (fm.region === "non-eu" && fm.recommendedAlternative) {
      const altId = serviceMap.get(fm.recommendedAlternative)
      const serviceId = serviceMap.get(slug)
      if (altId && serviceId) {
        await payload.update({
          collection: "services",
          id: serviceId,
          data: { recommendedAlternative: altId },
        })
      }
    }
  }

  console.log(`Imported ${serviceMap.size} services`)
  return serviceMap
}
```
